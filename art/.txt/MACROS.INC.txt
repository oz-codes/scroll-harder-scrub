15,01;#############################################################################
15,01; Memory Mode Register
15,01; in :
15,01;   AH = Bitplane of register values
15,01;       Bit 3 = C4 Chain Four
15,01;       Bit 2 = O/E Odd/Even
15,01;       Bit 1 = EM Extended Memory
15,01;       Bit 0 = Alpha/Graphics Mode
15,01; out :
15,01;   none
15,01;#############################################################################
15,01MemoryMode MACRO
15,01    mov     dx,3c4h                 ; Sequencer WRITE COLOR
15,01    mov     al,04h                  ; Index 04h - Memory Mode Register
15,01                                    ; value = 06h = 0000 0110b
15,01                                    ; bit 3 unchains the VGA
15,01    out     dx,ax
15,01ENDM MemoryMode

15,01;#############################################################################

15,01;       Bit 2 = EM2 Mask Memory Plane 2
15,01;       Bit 1 = EM1 Mask Memory Plane 1
15,01;       Bit 0 = EM0 Mask Memory Plane 0
15,01; out :
15,01;   none
15,01;#############################################################################
